367  T* allocate(std::size_t n) {
368  char* aligned_ptr = (char*)::operator new(alloc_size(n*sizeof(T)));
369  assert(is_max_aligned(aligned_ptr));
370 
371  char* unaligned_ptr = aligned_ptr + alignof(T);
372  assert(is_min_aligned(unaligned_ptr));
373 
374  P->countAlloc(unaligned_ptr, n * sizeof(T), alignof(T));
375 
376  return ((T*)unaligned_ptr);
377  }

379  void deallocate(T* p, std::size_t n) {
380  assert(is_min_aligned(p));
381 
382  char* aligned_ptr = ((char*)p) - alignof(T);
383  assert(is_max_aligned(aligned_ptr));
384 
385  P->countDealloc(p, n*sizeof(T), alignof(T));
386 
387  return ::operator delete(static_cast<void*>(aligned_ptr));
388  }

407  static std::size_t alloc_size(std::size_t s) {
408  std::size_t bytes = (s + BlockSize - 1) & ~(BlockSize - 1);
409  bytes += BlockSize;
410  assert(bytes % BlockSize == 0);
411  return bytes;
412  }

414  static bool is_max_aligned(void* p) {
415  return reinterpret_cast<std::uintptr_t>(p) % BlockSize == 0;
416  }
417 
418  static bool is_min_aligned(void* p) {
419  if (alignof(T) == BlockSize) {
420  return is_max_aligned(p);
421  } else {
422  return reinterpret_cast<std::uintptr_t>(p) % BlockSize == alignof(T);
423  }
