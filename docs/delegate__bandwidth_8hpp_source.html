115  eosio_assert( del.stake_cpu_quantity.amount >= 0, "must stake a positive amount" );
116  eosio_assert( del.stake_net_quantity.amount >= 0, "must stake a positive amount" );
117  eosio_assert( del.stake_storage_quantity.amount >= 0, "must stake a positive amount" );
118 
119  system_token_type total_stake = del.stake_cpu_quantity + del.stake_net_quantity + del.stake_storage_quantity;
120  eosio_assert( total_stake.quantity > 0, "must stake a positive amount" );
121 
122  require_auth( del.from );
123 
124  //eosio_assert( is_account( del.receiver ), "can only delegate resources to an existing account" );
125  uint64_t storage_bytes = 0;
126  if ( 0 < del.stake_storage_quantity.amount ) {
127  auto parameters = global_state_singleton::exists() ? global_state_singleton::get()
128  : common<SystemAccount>::get_default_parameters();
129  auto token_supply = currency::get_total_supply();
130 
131  //make sure that there is no posibility of overflow here
132  uint64_t storage_bytes_estimated = ( parameters.max_storage_size - parameters.total_storage_bytes_reserved )
133  * parameters.storage_reserve_ratio * system_token_type(del.stake_storage_quantity)
134  / ( token_supply - parameters.total_storage_stake ) / 1000 /* reserve ratio coefficient */;
135 
136  storage_bytes = ( parameters.max_storage_size - parameters.total_storage_bytes_reserved - storage_bytes_estimated )
137  * parameters.storage_reserve_ratio * system_token_type(del.stake_storage_quantity)
138  / ( token_supply - del.stake_storage_quantity - parameters.total_storage_stake ) / 1000 /* reserve ratio coefficient */;
139 
140  eosio_assert( 0 < storage_bytes, "stake is too small to increase storage even by 1 byte" );
141 
142  parameters.total_storage_bytes_reserved += storage_bytes;
143  parameters.total_storage_stake += del.stake_storage_quantity;
144  global_state_singleton::set(parameters);
145  }

214

216  parameters.total_storage_bytes_reserved -= del.unstake_storage_bytes;
217  parameters.total_storage_stake -= storage_stake_decrease;
