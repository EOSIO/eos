70  void newCalled(std::size_t s)
71  {
72  assert(disable_allocations == false);
73  assert(s);
74  if (throw_after == 0) {
75  throw_after = never_throw_value;
76  detail::throw_bad_alloc_helper();
77  } else if (throw_after != never_throw_value) {
78  --throw_after;
79  }
